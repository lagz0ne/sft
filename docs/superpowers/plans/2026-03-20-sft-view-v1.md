> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the experimental canvas with a shared review surface — screens list, screen detail with state machine strip, basic flow walkthrough. Minimal backend changes. No breaking changes to flow format, schema, or YAML.

**Architecture:** Two backend changes to `show.Spec` (add state list + parsed flow steps), then a frontend rebuild — routes replace overlays, strip replaces panel, tabs replace canvas. Reuse existing infrastructure: Go server, NATS, attachment serving, spec loader, TanStack Router, shadcn/ui.

**Deferred (needs explicit confirmation before starting):**
- New flow format (`starts` + `steps`) — breaking change to schema/parser/examples
- Snapshots — entirely new backend subsystem
- Maturity model — needs validator refactoring for entity scoping
- Mockup-per-state — schema migration for attachments
- Off-script walkthrough behavior — spec is intentionally loose

```go
func deriveStates(transitions []Transition) []string {
	if len(transitions) == 0 {
		return nil
	}
	seen := map[string]bool{}
	var ordered []string
	// First pass: collect from_state in order (first = initial)
	for _, t := range transitions {
		if t.FromState != "" && !seen[t.FromState] {
			seen[t.FromState] = true
			ordered = append(ordered, t.FromState)
		}
	}
	// Second pass: collect to_state that weren't from_state
	for _, t := range transitions {
		if t.ToState != "" && !seen[t.ToState] {
			seen[t.ToState] = true
			ordered = append(ordered, t.ToState)
		}
	}
	return ordered
}
```

In `Load()`, after loading flows, query `flow_steps` for each flow:

```go
// loadFlowSteps loads parsed steps for a flow by its DB id (not name, to avoid
// cross-app ambiguity). The `data` column is nullable — use sql.NullString.
func loadFlowSteps(db *sql.DB, flowID int64) ([]FlowStep, error) {
	rows, err := db.Query(`
		SELECT position, type, name, history, data
		FROM flow_steps
		WHERE flow_id = ?
		ORDER BY position`, flowID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var steps []FlowStep
	for rows.Next() {
		var s FlowStep
		var data sql.NullString
		if err := rows.Scan(&s.Position, &s.Type, &s.Name, &s.History, &data); err != nil {
			return nil, err
		}
		s.Data = data.String
		steps = append(steps, s)
	}
	return steps, nil
}
```

In the `Load()` function's flow loading loop, capture the flow ID and call `loadFlowSteps`:

```tsx
// web/apps/web/src/components/region-list.tsx
import type { Region } from '../lib/types'

function eventLabel(event: string): string {
  // "select_email(email)" → "Select email"
  const name = event.split('(')[0]
  return name.replace(/_/g, ' ').replace(/^\w/, c => c.toUpperCase())
}

export function RegionList({
  regions,
  visibleRegions,
}: {
  regions: Region[]
  visibleRegions?: string[] | null
}) {
  return (
    <div className="flex flex-col gap-1">
      {regions.map(region => {
        const hidden = visibleRegions && !visibleRegions.includes(region.name)
        return (
          <div
            key={region.name}
            className={`rounded-lg border border-neutral-200 bg-white p-3 ${hidden ? 'opacity-30' : ''}`}
          >
            <div className="font-semibold text-sm">{region.name}</div>
            {region.description && (
              <div className="text-xs text-neutral-500 mt-0.5">{region.description}</div>
            )}
            {region.events && region.events.length > 0 && (
              <div className="mt-1.5 flex flex-col gap-0.5">
                {region.events.map(event => (
                  <div key={event} className="text-xs text-neutral-400">
                    {eventLabel(event)}
                  </div>
                ))}
              </div>
            )}
            {/* Nested regions */}
            {region.regions && region.regions.length > 0 && (
              <div className="mt-2 ml-3 border-l border-neutral-100 pl-3">
                <RegionList regions={region.regions} visibleRegions={visibleRegions} />
              </div>
            )}
          </div>
        )
      })}
    </div>
  )
}
```

The strip shows states as chips, transitions as labeled edges. Clicking advances the state. Regions and mockup update via callback. Logic reused from the deleted `state-machine-panel.tsx`.

```tsx
// web/apps/web/src/components/state-machine-strip.tsx
import { useState, useCallback } from 'react'
import type { Transition } from '../lib/types'

function eventLabel(event: string): string {
  return event.replace(/_/g, ' ').replace(/^\w/, c => c.toUpperCase())
}

export function StateMachineStrip({
  states,
  transitions,
  onStateChange,
}: {
  states: string[]
  transitions: Transition[]
  onStateChange: (state: string) => void
}) {
  const [currentState, setCurrentState] = useState(states[0])

  const advance = useCallback((toState: string) => {
    setCurrentState(toState)
    onStateChange(toState)
  }, [onStateChange])

  const reset = useCallback(() => {
    setCurrentState(states[0])
    onStateChange(states[0])
  }, [states, onStateChange])

  // Transitions from current state
  const reachable = transitions.filter(t =>
    t.from_state === currentState || t.from_state === ''
  )

  // Forward transitions (non-self, non-navigate)
  const forwardEdges = reachable.filter(t =>
    t.to_state && t.to_state !== currentState && !t.action?.startsWith('navigate(')
  )

  // Self-transitions
  const selfTransitions = reachable.filter(t =>
    t.to_state === currentState || t.to_state === '.'
  )

  // Navigate-out transitions
  const navigateOuts = reachable.filter(t =>
    t.action?.startsWith('navigate(')
  )

  return (
    <div className="border border-neutral-200 rounded-lg bg-white p-3 mb-6">
      {/* State chips + edges */}
      <div className="flex items-center gap-1 overflow-x-auto">
        {states.map((state, i) => {
          const isCurrent = state === currentState
          const edge = forwardEdges.find(t => t.to_state === state)
          return (
            <div key={state} className="flex items-center gap-1 shrink-0">
              {/* Edge label before chip (if a forward edge leads here) */}
              {i > 0 && edge && (
                <>
                  <div className="w-4 h-px bg-neutral-900" />
                  <span className="text-[10px] text-neutral-500 whitespace-nowrap">
                    {eventLabel(edge.on_event)}
                  </span>
                  <div className="w-2 h-px bg-neutral-900" />
                  <div className="w-0 h-0 border-t-[3px] border-b-[3px] border-l-[5px] border-transparent border-l-neutral-900" />
                </>
              )}
              {i > 0 && !edge && (
                <>
                  <div className="w-6 h-px bg-neutral-200" />
                  <div className="w-0 h-0 border-t-[3px] border-b-[3px] border-l-[5px] border-transparent border-l-neutral-200" />
                </>
              )}
              {/* State chip */}
              <button
                className={`px-3 py-1 rounded text-sm whitespace-nowrap ${
                  isCurrent
                    ? 'border-2 border-neutral-900 font-semibold'
                    : 'border border-neutral-200 text-neutral-400'
                }`}
                onClick={() => {
                  // If there's a reachable edge to this state, advance
                  const t = reachable.find(tr => tr.to_state === state)
                  if (t) advance(state)
                }}
                disabled={!reachable.some(t => t.to_state === state) && !isCurrent}
              >
                {i === 0 && <span className="inline-block w-1.5 h-1.5 rounded-full bg-neutral-900 mr-1.5 align-middle" />}
                {state}
              </button>
            </div>
          )
        })}

        {/* Reset */}
        {currentState !== states[0] && (
          <button
            onClick={reset}
            className="ml-2 text-xs text-neutral-400 hover:text-neutral-600 shrink-0"
          >
            reset
          </button>
        )}
      </div>

      {/* Secondary: self-transitions + navigate-outs */}
      {(selfTransitions.length > 0 || navigateOuts.length > 0) && (
        <div className="flex gap-3 mt-2 text-xs text-neutral-400 flex-wrap">
          {selfTransitions.map(t => (
            <span key={t.on_event} className="cursor-pointer hover:text-neutral-600">
              ↻ {eventLabel(t.on_event)}
            </span>
          ))}
          {navigateOuts.map(t => {
            const target = t.action?.match(/navigate\((\w+)\)/)?.[1]
            return (
              <span key={t.on_event} className="cursor-pointer hover:text-neutral-600">
                ↗ {eventLabel(t.on_event)} → {target}
              </span>
            )
          })}
        </div>
      )}
    </div>
  )
}
```

- [ ] **Step 2: Create the flow walkthrough route**

```tsx
// web/apps/web/src/routes/flows.$name.tsx
import { createFileRoute, Link } from '@tanstack/react-router'
import { useSpecContext } from '../context/spec-context'
import { FlowStepStrip } from '../components/flow-step-strip'
import { RegionList } from '../components/region-list'
import { StateMachineStrip } from '../components/state-machine-strip'
import { useState } from 'react'

export const Route = createFileRoute('/flows/$name')({
  component: FlowWalkthrough,
})

function FlowWalkthrough() {
  const { name } = Route.useParams()
  const { spec } = useSpecContext()

  const flow = spec?.flows?.find(f => f.name === name)
  if (!flow) return <div className="p-8 text-neutral-400">Flow not found: {name}</div>

  const steps = flow.steps ?? []
  const [currentStepPos, setCurrentStepPos] = useState(steps[0]?.position ?? 1)

  const currentStep = steps.find(s => s.position === currentStepPos)

  // Find the screen for the current step
  // If step is a screen → that screen. If step is a region/event → find containing screen.
  const currentScreenName = findScreenForStep(currentStep, steps, currentStepPos, spec)
  const currentScreen = spec?.screens?.find(s => s.name === currentScreenName)

  const [machineState, setMachineState] = useState<string | null>(null)
  const effectiveState = machineState ?? currentScreen?.states?.[0] ?? null
  const visibleRegions = effectiveState && currentScreen?.state_regions
    ? currentScreen.state_regions[effectiveState]
    : null

  const canNext = currentStepPos < (steps[steps.length - 1]?.position ?? 0)
  const canBack = currentStepPos > (steps[0]?.position ?? 0)

  return (
    <div className="max-w-3xl mx-auto p-8">
      <Link to="/" className="text-sm text-neutral-400 hover:text-neutral-600 mb-4 block">
        ← back
      </Link>

      <h1 className="text-xl font-bold mb-1">{flow.name}</h1>
      {flow.description && (
        <p className="text-neutral-500 mb-4 text-sm">{flow.description}</p>
      )}

      {/* Step strip */}
      <FlowStepStrip
        steps={steps}
        currentStep={currentStepPos}
        onStepClick={setCurrentStepPos}
      />

      {/* Current screen detail */}
      {currentScreen && (
        <div className="border border-neutral-200 rounded-lg bg-white p-5">
          <div className="font-semibold text-lg mb-4">{currentScreen.name}</div>

          {/* State machine strip if applicable */}
          {currentScreen.states && currentScreen.states.length > 1 && (
            <StateMachineStrip
              states={currentScreen.states}
              transitions={currentScreen.transitions ?? []}
              onStateChange={setMachineState}
            />
          )}

          {/* Regions */}
          {currentScreen.regions && currentScreen.regions.length > 0 && (
            <RegionList regions={currentScreen.regions} visibleRegions={visibleRegions} />
          )}
        </div>
      )}

      {/* Navigation */}
      <div className="flex gap-3 mt-4">
        <button
          onClick={() => {
            const idx = steps.findIndex(s => s.position === currentStepPos)
            if (idx > 0) {
              setCurrentStepPos(steps[idx - 1].position)
              setMachineState(null)
            }
          }}
          disabled={!canBack}
          className="px-4 py-2 border border-neutral-200 rounded-lg text-sm disabled:opacity-30"
        >
          ← Back
        </button>
        <button
          onClick={() => {
            const idx = steps.findIndex(s => s.position === currentStepPos)
            if (idx < steps.length - 1) {
              setCurrentStepPos(steps[idx + 1].position)
              setMachineState(null)
            }
          }}
          disabled={!canNext}
          className="px-4 py-2 border border-neutral-200 rounded-lg text-sm disabled:opacity-30"
        >
          Next step →
        </button>
      </div>
    </div>
  )
}

// Helper: find the most recent screen step at or before the current position
function findScreenForStep(
  currentStep: { type: string; name: string } | undefined,
  steps: { position: number; type: string; name: string }[],
  currentPos: number,
  spec: any
): string | undefined {
  if (currentStep?.type === 'screen') return currentStep.name

  // Walk backwards to find the last screen step
  for (let i = steps.length - 1; i >= 0; i--) {
    if (steps[i].position <= currentPos && steps[i].type === 'screen') {
      return steps[i].name
    }
  }

  // Fallback: check if the step name is a region, find its parent screen
  if (currentStep) {
    for (const screen of spec?.screens ?? []) {
      if (findRegionInScreen(screen.regions, currentStep.name)) {
        return screen.name
      }
    }
  }
  return undefined
}

function findRegionInScreen(regions: any[] | undefined, name: string): boolean {
  if (!regions) return false
  for (const r of regions) {
    if (r.name === name) return true
    if (findRegionInScreen(r.regions, name)) return true
  }
  return false
}
```
